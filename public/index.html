import IlyassAI from 'ilyassai';

const client = new IlyassAI({
  apiKey: process.env.ILYASS_API_KEY
});

const response = await client.chat.complete({
  model: 'ilyass-3-turbo',
  messages: [{
    role: 'user',
    content: 'Explain quantum computing'
  }],
  stream: true
});

// Stream the response
for await (const chunk of response) {
  process.stdout.write(chunk.text);
